 17    @staticmethod
 18    def create_tokens(username: str, token_type: str) -> str:
 19        """
 20        Generate an access or refresh token for the user.
 21
 22        The token is created with an expiration time based on the type
 23        (either "access" or "refresh") and encoded with a secret key.
 24
 25        Parameters
 26        ----------
 27            username (str): The username for whom the token is created.
 28            token_type (str): The type of token to create, either "access" or "refresh".
 29
 30        Returns
 31        -------
 32            str: The generated JSON Web Token (JWT).
 33        """
 34
 35        if token_type == "access":
 36            minutes = os.getenv("ACCESS_TOKEN_EXPIRATION_MINUTES")
 37        else:
 38            minutes = os.getenv("REFRESH_TOKEN_EXPIRATION_MINUTES")
 39        payload = {}
 40        if minutes:
 41            payload = {
 42                "exp": datetime.datetime.utcnow()
 43                + datetime.timedelta(minutes=int(minutes)),
 44                "sub": username,
 45                "type": token_type,
 46            }
 47        token = jwt.encode(payload, os.getenv("SECRET_KEY"), algorithm="HS256")
 48        Auth.write_token_hash_to_db(
 49            token=token, username=username, token_type=token_type
 50        )
 51        return token

 53    @staticmethod
 54    def write_token_hash_to_db(token: str, username: str, token_type: str) -> None:
 55        """
 56        Hash and store a token in the database for the specified user.
 57
 58        Parameters
 59        ----------
 60            token (str): The token to hash and store.
 61            username (str): The username associated with the token.
 62            token_type (str): The type of token, either "access" or "refresh".
 63
 64        Returns
 65        -------
 66            None
 67        """
 68        user = Users.query.filter_by(username=username).first()
 69        if user:
 70            if token_type == "access":
 71                user.access_token_hash = generate_password_hash(token)
 72            else:
 73                user.refresh_token_hash = generate_password_hash(token)
 74            db.session.commit()

 76    @staticmethod
 77    def check_access_token(token: str, username: str, token_data) -> bool:
 78        """
 79        Verify the validity of a user's access token.
 80
 81        Checks the database for the hashed token and verifies its type.
 82
 83        Parameters
 84        ----------
 85            token (str): The token to check.
 86            username (str): The username associated with the token.
 87            token_data: Decoded token data, including type information.
 88
 89        Returns
 90        -------
 91            bool: True if the token is valid. False otherwise.
 92        """
 93        user = Users.query.filter_by(username=username).first()
 94        if user and user.refresh_token_hash:
 95            if (
 96                check_password_hash(user.refresh_token_hash, token)
 97                and token_data["type"] == "refresh"
 98            ):
 99                return True
100            else:
101                return False
102        return False

104    @staticmethod
105    def token_required(f):
106        """
107        Decorator function that ensures a valid access token is provided.
108
109        Checks the provided token's validity and type before allowing access to the
110        decorated route.
111
112        Parameters
113        ----------
114            f: The function to decorate.
115
116        Returns
117        -------
118            The decorated function or a response indicating the token issue.
119        """
120
121        @wraps(f)
122        def decorated(*args, **kwargs):
123            token = request.headers.get("jasonWebToken")
124            if token:
125                try:
126                    token_data = jwt.decode(
127                        token, os.getenv("SECRET_KEY"), algorithms=["HS256"]
128                    )
129                    current_user = token_data["sub"]
130                    user = Users.query.filter_by(username=current_user).first()
131                    if user:
132                        if token_data["type"] != "access" or not check_password_hash(
133                            user.access_token_hash, token
134                        ):
135                            return (
136                                "Invalid access token",
137                                403,
138                            )
139
140                except jwt.ExpiredSignatureError:
141                    return "Access token expired. Please use refresh token", 403
142                except jwt.InvalidTokenError:
143                    return "Invalid access token. Please use refresh token", 403
144            else:
145                return "Access token required", 403
146            return f(*args, current_user, **kwargs)
147
148        return decorated

151    def login_user(username: str, password: str) -> bool:
152        """
153        Log in a user by validating their credentials.
154
155        Checks the password for the given username and updates the last login timestamp.
156
157        Parameters
158        ----------
159            username (str): The username of the user logging in.
160            password (str): The plain-text password for validation.
161
162        Returns
163        -------
164            bool: True if login is successful. False if credentials are invalid.
165        """
166        user = Users.query.filter_by(username=username).first()
167        if user and check_password_hash(user.password_hash, password):
168            user.last_login = datetime.datetime.utcnow()
169            db.session.commit()
170            return True
171        else:
172            return False
